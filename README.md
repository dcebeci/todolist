#Dependencies

Springboot v3.3.2,
JDK 21.0.4,
React 18.3.1,
MySQL 8.0


#Setup 

git clone https://github.com/dcebeci/todolist.git

cd todolist

#Backend startup

./mvnw clean install

./mvnw spring-boot:run

#Frontend Startup

cd todolist-frontend

npm install

npm start
